 select CONCAT(FirstName,' ',left(Lastname,1)) as "Nombre", Sum([Order Details].Quantity) as "venta totales" 
 from [Order Details] inner join Orders on [Order Details].OrderID=Orders.OrderID
             inner join Employees on Orders.EmployeeID=Orders.EmployeeID
 group by FirstName,LastName
 having sum(Quantity)>100000
